//COMMAND//	//PARAMETER//		//WHO//		//DESCRIPTION//				//NOTES//
!quit					#OWNER#		Kills the bot

!lock					#ADMIN#		Lock a game
!unlock					#ADMIN#		Unlock a game
!getdata				#ADMIN#		Provides ip + pw to the admin
!addban		<nick>			#ADMIN#		Add a ban for the nickname 		(nickname-based)
!delban		<nick>			#ADMIN#		Remove a ban for the nickname
!addmap		<map>			#ADMIN#		Add a new map				!!WITHOUT ut4_!!
!delmap		<map>			#ADMIN#		Remove a map				!!WITHOUT ut4_!!
!setserver	<ip:port>		#ADMIN#		Set the server
!setrcon	<rconpassword>		#ADMIN#		Set the rconpassword
!test		<0-9>			#ADMIN#		Set the player number			FOR TEST PROPOSES ONLY

!reset					#OP#		Resets a game
!rcon 		<cmd>			#OP#		Send a rcon cmd to the server

!remove					#ADDED#		Remove you from pickup
!maps					#ADDED#		Show the available maps
!map <ut4_/map>				#ADDED#		Vote for a map				!!with or without ut4_!!
!lostpass				#ADDED#		Request the pw of the gameserver
!gameover				#ADDED#		Submit a gameover request - 4 needed
!ring					#ADDED#		Spam a !ringer msg

!add					#USER#		Add you to pickup
!status					#USER#		Show who is added
!help					#USER#		Show the available pickup cmd list
